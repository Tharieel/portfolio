import pandas as pd
            import plotly.graph_objects as go
            from plotly.subplots import make_subplots
            from sklearn.cluster import DBSCAN
            from scipy.spatial import ConvexHull
            import numpy as np

            # ------------------------------
            # CSV/TSV laden
            # ------------------------------
            df = pd.read_csv("raw/Occurrence.tsv", sep='\t', low_memory=False)
            df = df.dropna(subset=['scientificName', 'eventDate', 'decimalLatitude', 'decimalLongitude'])
            df['eventDate'] = pd.to_datetime(df['eventDate'])
            df['year'] = df['eventDate'].dt.year

            # Mapping wissenschaftlicher Name → deutscher Name
            name_map = {
                "Larus novaehollandiae": "Silbermöwe",
                "Sterna bergii": "Weißbrustseeschwalbe",
                "Egretta sacra": "Rotschnabelreiher",
                "Sterna sumatrana": "Sumatra-Seeschwalbe",
                "Sula leucogaster": "Braunschwingen-Tölpel",
                "Anous minutus": "Schwarzfußtölpel",
                "Onychoprion anaethetus": "Graukopftölpel",
                "Anous stolidus": "Noddi-Tölpel",
                "Fregata ariel": "Fregattvogel",
                "Sterna bengalensis": "Bengalseeschwalbe"
            }
            df["scientificName"] = df["scientificName"].apply(
                lambda x: f"{x} ({name_map[x]})" if x in name_map else x
            )

            # ------------------------------
            # DBSCAN Clustering
            # ------------------------------
            coords = df[['decimalLatitude', 'decimalLongitude']].to_numpy()
            clustering = DBSCAN(eps=1, min_samples=2).fit(coords)
            df = pd.concat([df, pd.Series(clustering.labels_, name='cluster')], axis=1)
            df['cluster_label'] = df['cluster'].apply(lambda x: 'Noise' if x == -1 else f'Hotspot {x}')

            # ------------------------------
            # Subplot: Scatter + Heatmap-Inset
            # ------------------------------

            fig = make_subplots()

            # Hauptscatter: SST-farbig (nur SST bekommt Colorbar!)
            scatter_trace = go.Scatter(
                x=df["decimalLongitude"],
                y=df["decimalLatitude"],
                mode="markers",
                marker=dict(
                    color=df["sst"],
                    colorscale="Plasma",
                    size=8,
                    line=dict(width=0),
                    colorbar=dict(
                        title="SST (°C)",
                        title_side="right",
                        tickfont=dict(size=12),
                        lenmode="fraction",
                        len=1,
                        thickness=20
                    )
                ),
                hovertext=df["scientificName"],
                hoverinfo="text",
                showlegend=False  # Keine Legende für SST
            )

            fig.add_trace(scatter_trace)

            # Cluster-Hüllkurven hinzufügen, ohne Einfluss auf Colorbar
            hotspot_labels = [l for l in df['cluster_label'].unique() if l != 'Noise']
            for label in hotspot_labels:
                cluster_points = df[df['cluster_label'] == label][['decimalLongitude','decimalLatitude']].to_numpy()
                if len(cluster_points) < 3:
                    continue
                if np.linalg.matrix_rank(cluster_points - cluster_points[0]) < 2:
                    continue
                hull = ConvexHull(cluster_points)
                hull_points = cluster_points[hull.vertices]
                hull_points = np.append(hull_points, [hull_points[0]], axis=0)
                fig.add_trace(go.Scatter(
                    x=hull_points[:,0],
                    y=hull_points[:,1],
                    fill='toself',
                    fillcolor='rgba(0,0,255,0.05)',  # leicht transparent blau
                    line=dict(color='rgba(0,0,0,0)'),
                    name=f'{label} Area',
                    hoverinfo='skip',
                    showlegend=False             # Kein Eintrag in Legende
                ))

            # Heatmap-Inset links unten (nur Dichte, keine SST)
            heatmap = go.Histogram2d(
                x=df["decimalLongitude"],
                y=df["decimalLatitude"],
                colorscale="Viridis",
                opacity=0.8,
                showscale=False
            )
            fig.add_trace(heatmap)
            fig.data[-1].update(xaxis="x2", yaxis="y2")

            # ------------------------------
            # Layout Haupt- und Inset-Achse
            # ------------------------------
            fig.update_layout(
                plot_bgcolor="rgba(46,77,144,0.3)",
                title=dict(
                    text="Seabird Observations – Great Barrier Reef 1991-1996 (Interactive)",
                    x=0.5,
                    font=dict(size=20)
                ),
                xaxis=dict(domain=[0, 1], title="Longitude (Längengrad)", tickfont=dict(size=14)),
                yaxis=dict(domain=[0, 1], title="Latitude (Breitengrad)", tickfont=dict(size=14)),

                # Inset-Achse Heatmap links unten
                xaxis2=dict(domain=[0.02, 0.28], anchor="y2", showgrid=False, showticklabels=False),
                yaxis2=dict(domain=[0.02, 0.28], anchor="x2", showgrid=False, showticklabels=False),

                width=900,
                height=600,
                legend_title_text='Cluster'
            )

            fig.show()

            fig.write_html("plots/seabird_combined_interactive.html")
